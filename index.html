quickstart.js 
// Initialize MuxBite client
import MuxBite from '@muxbite/sdk';

const mb = new MuxBite({
  apiKey: process.env.MB_API_KEY
});

// Create a new live stream
const stream = await mb.streams.create({
  playback: 'public',
  latency: 'low',
  resolution: '1080p'
});

// Returns stream key instantly
console.log(stream.key);
// → "mb_sk_7x9f2a..."
